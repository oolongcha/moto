Example Usage


<script>
  var photoSlider = new WallopSlider('.photo-slider');
  var secondarySlider = new WallopSlider('.secondary-slider');

  photoSlider.on('change', function (event) {
    alert('change event dispatched from ' + event.detail.parentSelector + ' / Current item index: ' + event.detail.currentItemIndex);
  });

  secondarySlider.on('change', function (event) {
    alert('change event dispatched from ' + event.detail.parentSelector + ' / Current item index: ' + event.detail.currentItemIndex);
  });
</script>
